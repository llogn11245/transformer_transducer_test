.. code-block:: bash

    speeq
    ├── config
    ├── constants
    ├── data
    │   ├── augmenters
    │   ├── decorators
    │   ├── loaders
    │   ├── padders
    │   ├── processes
    │   ├── processors
    │   ├── registry
    │   └── tokenizers
    ├── interfaces
    ├── models
    │   ├── activations
    │   ├── ctc
    │   ├── decoders
    │   ├── encoders
    │   ├── layers
    │   ├── registry
    │   ├── seq2seq
    │   ├── skeletons
    │   ├── templates
    │   └── transducers
    ├── predictors
    ├── trainers
    │   ├── criterions
    │   ├── decorators
    │   ├── registry
    │   ├── schedulers
    │   ├── templates
    │   └── trainers
    ├── utils
    │   ├── loggers
    │   └── utils
    └── _version
